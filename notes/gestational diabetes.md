```
def selROC(df,fpr=.1,tpr=None,th=None,eps=0.01,p=0.017,ppv=False):
    row=df[df.fpr.between(fpr-eps,fpr+eps)].max()
    if ppv:
        ppv=row.tpr/(row.tpr+row.fpr*((1./p - 1)))
        return row,ppv
    else:
        return row.tpr
def getPPV(row,RHO):
    return row.tpr/(row.tpr + row.fpr*(1./RHO -1))
```
